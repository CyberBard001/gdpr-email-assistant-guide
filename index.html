function logUniqueSendersToSheet() {
  const threads = GmailApp.search("newer_than:1y");
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Senders");
  const data = sheet.getDataRange().getValues().map(row => row[0]);
  const now = new Date();

  threads.forEach(thread => {
    const sender = thread.getMessages()[0].getFrom();
    const domain = sender.substring(sender.indexOf("@") + 1).toLowerCase();
    if (!data.includes(sender)) {
      sheet.appendRow([sender, now, "Pending", "", "", domain]);
    }
  });
}

function extractDomains() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Senders");
  const data = sheet.getDataRange().getValues();

  for (let i = 1; i < data.length; i++) {
    const email = data[i][0];
    const domain = email.substring(email.indexOf("@") + 1).toLowerCase();
    sheet.getRange(i + 1, 6).setValue(domain);
  }
}

function cleanupDuplicatesByDomain() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Senders");
  const data = sheet.getDataRange().getValues();
  const seen = new Set();

  for (let i = 1; i < data.length; i++) {
    const domain = data[i][5];
    if (seen.has(domain)) {
      sheet.deleteRow(i + 1);
      i--;
    } else {
      seen.add(domain);
    }
  }
}

function sendGDPRRequests() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Senders");
  const data = sheet.getDataRange().getValues();

  for (let i = 1; i < data.length; i++) {
    const status = data[i][2];
    if (status === "Approved") {
      const email = data[i][0];
      GmailApp.sendEmail(email, "GDPR Data Request", 
        "Under the UK GDPR, I am requesting a copy of all personal data you hold about me and information about how it is processed. Please respond within the statutory period.");
      sheet.getRange(i + 1, 3).setValue("Sent");
    }
  }
}

function onOpen() {
  const ui = SpreadsheetApp.getUi();
  ui.createMenu("GDPR Tools")
    .addItem("Scan for new senders", "logUniqueSendersToSheet")
    .addItem("Extract domains", "extractDomains")
    .addItem("Clean duplicates by domain", "cleanupDuplicatesByDomain")
    .addItem("Send GDPR requests", "sendGDPRRequests")
    .addToUi();
}

‚úÖ Step 3: First-Time Manual Run

  • Run logUniqueSendersToSheet
  • Run extractDomains
  • Run cleanupDuplicatesByDomain
  • Mark rows in the sheet as Approved, Ignore, or Removed
  • Run sendGDPRRequests to email approved senders

‚úÖ Step 4: Use the Custom Menu

Once the script is run, a new menu will appear in your sheet called GDPR Tools. It will let you run everything with a click.
